17  package org.apache.commons.io.filefilter;
18  
19  import java.io.File;
20  
21  /**
22   * This filter accepts <code>File</code>s that are files (not directories).
23   * <p>
24   * For example, here is how to print out a list of the real files
25   * within the current directory:
26   *
27   * <pre>
28   * File dir = new File(".");
29   * String[] files = dir.list( FileFileFilter.FILE );
30   * for ( int i = 0; i &lt; files.length; i++ ) {
31   *     System.out.println(files[i]);
32   * }
33   * </pre>
34   *
35   * @since Commons IO 1.3
36   * @version $Revision: 155419 $ $Date: 2006-08-28 08:39:07 +0200 (Mo, 28 Aug 2006) $
37   */
38  public class FileFileFilter extends AbstractFileFilter {
39  
40      /** Singleton instance of file filter */
41      public static final IOFileFilter FILE = new FileFileFilter();
42  
43      /**
44       * Restrictive consructor.
45       */
46      protected FileFileFilter() {
47      }
48  
49      /**
50       * Checks to see if the file is a file.
51       *
52       * @param file  the File to check
53       * @return true if the file is a file
54       */
55      public boolean accept(File file) {
56          return file.isFile();
57      }
58  
59  }
